(
    ID SERIAL PRIMARY KEY,
    ISSUESLIST_ID BIGINT UNSIGNED NOT NULL,
    SUMMARY VARCHAR(100),
    DESCRIPTION VARCHAR(1024),
    FOREIGN KEY (ISSUESLIST_ID) REFERENCES USERS(ID)
);
